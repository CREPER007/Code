#include <stdio.h>
#include <stdlib.h>

#define SIZE 10

// Node structure
struct node {
    int key;
    struct node* next;
};

// Hash table
struct node* table[SIZE];

// Hash function
int hash(int key) {
    return key % SIZE;
}

// Insert key
void insert(int key) {
    int index = hash(key);

    struct node* newnode = (struct node*)malloc(sizeof(struct node));
    newnode->key = key;
    newnode->next = table[index];

    table[index] = newnode;
}

// Search key
void search(int key) {
    int index = hash(key);
    struct node* temp = table[index];

    while (temp) {
        if (temp->key == key) {
            printf("Key found\n");
            return;
        }
        temp = temp->next;
    }
    printf("Key not found\n");
}

// Display table
void display() {
    for (int i = 0; i < SIZE; i++) {
        struct node* temp = table[i];
        printf("%d: ", i);

        while (temp) {
            printf("%d -> ", temp->key);
            temp = temp->next;
        }
        printf("NULL\n");
    }
}

// Main
int main() {
    int choice, key;

    // Initialize table
    for (int i = 0; i < SIZE; i++)
        table[i] = NULL;

    while (1) {
        printf("\n1.Insert 2.Search 3.Display 4.Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);

        switch (choice) {
            case 1:
                printf("Enter key: ");
                scanf("%d", &key);
                insert(key);
                break;

            case 2:
                printf("Enter key: ");
                scanf("%d", &key);
                search(key);
                break;

            case 3:
                display();
                break;

            case 4:
                exit(0);

            default:
                printf("Invalid choice\n");
        }
    }
}
